signal-computed works!

  counterA = signal(0);
  counterB = signal(0);
  counterC = computed(() => this.counterA() + 2*this.counterB());

  clickedA() {
    this.counterA.update((prev) => prev + 1);
  }
  clickedB() {
    this.counterB.update((prev) => prev + 10);
  }
